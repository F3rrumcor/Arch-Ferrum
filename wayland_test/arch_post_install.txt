# Sync clock
sudo timedatectl set-ntp true

# Disable pc speaker
sudo nano /etc/modprobe.d/nobeep.conf
^
blacklist pcspkr

# Enable multilib
sudo nano /etc/pacman.conf

# Hibernation
sudo nano /etc/mkinitcpio.conf HOOKS=(base udev autodetect keyboard modconf block filesystems resume fsck) # resume after udev
sudo mkinitcpio -p linux
sudo nano /boot/loader/entries/arch.conf (resume=UUID=)
sudo bootctl update

# Install AUR package manager yay
sudo pacman -S git
cd /opt
sudo git clone https://aur.archlinux.org/yay-git.git
sudo chown -R user:group ./yay-git
cd yay-git
makepkg -si
sudo pacman -R go

# Display driver (intel)
sudo pacman -S --needed mesa lib32-mesa vulkan-intel lib32-vulkan-intel vulkan-icd-loader lib32-vulkan-icd-loader

# Window manager (hyprland)
yay -S hyprland-bin
yay -S xdg-desktop-portal-hyprland-git

# enable seatd
sudo systemctl enable seatd
sudo systemctl start seatd

# install sunshine
yay -S sunshine
echo 'KERNEL=="uinput", GROUP="input", MODE="0660", OPTIONS+="static_node=uinput"' | sudo tee /etc/udev/rules.d/85-sunshine-input.rules
mkdir -p ~/.config/systemd/user/
sudo nano ~/.config/systemd/user/sunshine.service
^
[Unit]
Description=Sunshine Gamestream Server for Moonlight
After=network.target

[Service]
Restart=always
RestartSec=30
ExecStart=/usr/bin/sunshine

[Install]
WantedBy=default.target
---
systemctl --user enable sunshine
sudo setcap cap_sys_admin+p $(readlink -f $(which sunshine))

# auto login
cp /usr/lib/systemd/system/getty@.service /etc/systemd/system/autologin@.service
ln -s /etc/systemd/system/autologin@.service /etc/systemd/system/getty.target.wants/getty@tty1.service
sudo nano /etc/systemd/system/autologin@.service < ExecStart=-/sbin/agetty -a USERNAME %I 38400
sudo systemctl daemon-reload
nano ~/.bash_profile
^
if [[ -z $DISPLAY ]] && [[ $(tty) = /dev/tty1 ]]; then
    exec hyprland
fi

# Audio (pipewire)
sudo pacman -S wireplumber pipewire pipewire-pulse pipewire-alsa pipewire-jack pavucontrol

# File management
sudo pacman -S thunar thunar-volman gvfs gvfs-smb ntfs-3g p7zip xarchiver

# Terminal emulator (alacritty)
sudo pacman -S alacritty

# Text editor (gedit)
sudo pacman -S gedit

# Browser (firefox)
sudo pacman -S firefox

# Video player (celluloid)
sudo pacman -S mpv celluloid

# Layan Cursors
mkdir -p $HOME/build
cd "$HOME/build"
git clone https://github.com/vinceliuice/Layan-cursors
cd Layan-cursors
sudo ./install.sh

# Nordic-Theme
yay -S nordic-theme

# Steam
sudo pacman -S steam

# Gamemode
sudo pacman -S gamemode lib32-gamemode

# Lutris
sudo pacman -S lutris vkd3d lib32-vkd3d
