# Sync clock
sudo timedatectl set-ntp true

# Enable multilib
sudo nano /etc/pacman.conf

# Kernel modules
sudo pacman -S dkms linux-headers
sudo dkms autoinstall

# Hibernation
sudo nano /etc/mkinitcpio.conf HOOKS=(base udev autodetect keyboard modconf block filesystems resume fsck) # resume after udev
sudo mkinitcpio -p linux
sudo nano /boot/loader/entries/arch.conf (resume=UUID=)
bootctl update

# Display driver (nvidia)
sudo pacman -S --needed nvidia-dkms nvidia-utils lib32-nvidia-utils nvidia-settings vulkan-icd-loader lib32-vulkan-icd-loader

# Display server (xorg)
sudo pacman -S xorg-server
localectl set-x11-keymap de

# Display manager (sddm)
sudo pacman -S sddm
sudo systemctl enable sddm
sudo systemctl start sddm
sudo nano /usr/share/sddm/scripts/Xsetu (xrandr --output DP-0 --primary --mode 1920x1080 --rate 143.98 --pos 0x0 --rotate normal --output DP-1 --off --output HDMI-0 --off --output HDMI-1 --off --output DVI-D-0 --off)
sudo cp /usr/lib/sddm/sddm.conf.d/default.conf /etc/sddm.conf
sudo nano /etc/sddm.conf

# Window manager (qtile)
sudo pacman -S qtile picom nitrogen lxsession lxappearance

# Numlock on
/etc/sddm.conf

[General]
...
Numlock=on

# Keepass

# Install AUR package manager yay

sudo git clone https://aur.archlinux.org/yay-git.git
sudo chown -R user:group ./yay-git
cd yay-git
makepkg -si
sudo pacman -R go
